Erstellen Sie einen Kellerautomaten, der folgende Sprache
\index{Kellerautomat}
\footcite[Seite 23]{theo:fs:2}

\begin{center}
\bAusdruck{a^n b^n}{n \in \mathbb{N}}
\end{center}

\noindent
mit folgender Grammatik

\begin{center}
\bGrammatik{
  variablen=S,
  alphabet={a,b},
  produktionen={S -> aSb | ab}
}
\end{center}

\noindent
erkennt.

  zustaende={\z1, \z2},
  alphabet={a, b},
  kelleralphabet={\#, A},
  start=\z1,
  ende={\z2},
}
\end{center}
