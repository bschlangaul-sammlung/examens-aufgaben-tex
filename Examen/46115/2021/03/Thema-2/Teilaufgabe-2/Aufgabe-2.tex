\begin{minted}{java}
class Stack {
  private Item head;

  public Stack() {
    head = null;
  }

  public void push(int val) {
    if (head == null) {
      head = new Item(val, null);
    } else {
      head = new Item(val, head);
    }
  }

  public int pop() {
    // ...
  }

  public int size() {
    // ...
  }

  public int min() {
    // ...
  }

  class Item {
    private int val;
    private Item next;

    public Item(int val, Item next) {
      this.val = val;
      this.next = next;
    }
  }
}
\end{minted}
